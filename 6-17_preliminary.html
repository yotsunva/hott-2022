{-# OPTIONS --without-K --safe #-}

id : ∀ {ℓ} {A : Set ℓ} → (A → A)
id a = a

_∘_ : ∀ {ℓ} {A B C : Set ℓ} → (B → C) → (A → B) → (A → C)
(g ∘ f) a = g (f a)

data _≡_ {ℓ} {A : Set ℓ} (a : A) : A → Set ℓ where
  refl : a ≡ a
infix 4 _≡_

ap : ∀ {ℓ} {A B : Set ℓ} (f : A → B) {a b : A} → a ≡ b → f a ≡ f b
ap f refl = refl

_⁻¹ : ∀ {ℓ} {A : Set ℓ} {a b : A} → a ≡ b → b ≡ a
refl ⁻¹ = refl

_∙_ : ∀ {ℓ} {A : Set ℓ} {a b c : A} → a ≡ b → b ≡ c → a ≡ c
refl ∙ p = p
infix 10 _∙_

record Σ {ℓ} (B : Set ℓ) (E : B → Set ℓ) : Set ℓ where
  constructor _,_
  field
    fst : B
    snd : E fst
open Σ public
infix 4 _,_
syntax Σ B (λ x → E-x) = Σ x ꞉ B , E-x

_×_ : ∀ {ℓ} (A B : Set ℓ) → Set ℓ
A × B = Σ a ꞉ A , B

_~_ : ∀ {ℓ} {A B : Set ℓ} (f g : A → B) → Set ℓ
_~_ {A = A} f g = (x : A) → f x ≡ g x
infix 4 _~_

~refl : ∀ {ℓ} {A B : Set ℓ} {f : A → B} → f ~ f
~refl x = refl

isheq : ∀ {ℓ} {A B : Set ℓ} (f : A → B) → Set ℓ
isheq {A = A} {B = B} f = Σ g ꞉ (B → A) , ((f ∘ g ~ id) × (g ∘ f ~ id))

iscontr : ∀ {ℓ} → Set ℓ → Set ℓ
iscontr A = Σ a ꞉ A , ((x : A) → a ≡ x)
