{-# OPTIONS --without-K --safe #-}

open import 6-17_preliminary

-- hfiber
_⁻¹_ : ∀ {ℓ} {A B : Set ℓ} (f : A → B) (b : B) → Set ℓ
_⁻¹_ {A = A} f b = Σ x ꞉ A , (f x ≡ b)
-- 「始域の点 x」と「x が (homotopy) fiber になっていることの証明」の組がなす型

-- hfiberact
fibact : ∀ {ℓ} {A B C : Set ℓ} {f : A → B} {g : C → B}
  (h : A → C) (H : g ∘ h ~ f) (b : B)
  → (f ⁻¹ b) → (g ⁻¹ b)
fibact h H b (a , p) = h a , (H a) ∙ p
-- h : A → C が fiber 間の関数 (f ⁻¹ b) → (g ⁻¹ b) を誘導する

-- maponhfiber
maponfib : ∀ {ℓ} {A B C : Set ℓ} (f : A → B) (g : B → C) (c : C)
  → ((g ∘ f) ⁻¹ c) → (g ⁻¹ c)
maponfib f g = fibact f (~refl {f = (g ∘ f)})
-- fibact で f = g ∘ f , g = g , h = f , H = (constant な homotopy) としたもの
-- f : A → C が fiber 間の関数 ((g ∘ f) ⁻¹ b) → (g ⁻¹ b) を誘導する

-- secmaponhfiber
secmaponfib : ∀ {ℓ} {A B : Set ℓ} {r : A → B} {s : B → A}
  (H : r ∘ s ~ id) (a : A)
  → (s ⁻¹ a) → ((s ∘ r) ⁻¹ a)
secmaponfib {s = s} H a = fibact s (λ x → ap s (H x)) a
-- r の section s に対して、
-- fibact で f = s , g = s ∘ r , h = s , H = λ x → ap s (H x) としたもの

-- secmaponhfiberissec
secmaponfibsec : ∀ {ℓ} {A B : Set ℓ} {r : A → B} {s : B → A}
  (H : r ∘ s ~ id) (a : A)
  → (maponfib r s a) ∘ (secmaponfib H a) ~ id
secmaponfibsec {B = B} {s = s} H .(s b) (b , refl) = lemma (H b)
  where
  lemma : {rsb b : B} (Hb : rsb ≡ b) → (rsb , refl ∙ ((ap s Hb) ∙ refl)) ≡ (b , refl)
  lemma refl = refl

-- hfiberandhomot
fib2fib : ∀ {ℓ} {A B : Set ℓ} {f g : A → B} (b : B) (H : f ~ g)
  → (f ⁻¹ b) → (g ⁻¹ b)
fib2fib b H = fibact id (λ a → (H a) ⁻¹) b
-- fibact で h = id としたもの（H をひっくり返して使う）

-- hfiberandhomotinv
fib2fib⁻¹ : ∀ {ℓ} {A B : Set ℓ} {f g : A → B} (b : B) (H : f ~ g)
  → (g ⁻¹ b) → (f ⁻¹ b)
fib2fib⁻¹ b H = fibact id H b
-- fibact で h = id としたもの（H をそのまま使う）

-- isheqhfiberandhomot
fibheq : ∀ {ℓ} {A B : Set ℓ} {f g : A → B} (b : B) (H : f ~ g)
  → isheq (fib2fib b H)
fibheq b H .fst = fib2fib⁻¹ b H
fibheq {B = B} {g = g} .(g a) H .snd .fst (a , refl) = ap (λ x → (a , x)) (lemma (H a))
  where
  lemma : {fa ga : B} (Ha : fa ≡ ga) → Ha ⁻¹ ∙ (Ha ∙ refl) ≡ refl
  lemma refl = refl
fibheq {B = B} {f = f} .(f a) H .snd .snd (a , refl) = ap (λ x → (a , x)) (lemma (H a))
  where
  lemma : {fa ga : B} (Ha : fa ≡ ga) → Ha ∙ (Ha ⁻¹ ∙ refl) ≡ refl
  lemma refl = refl

-- isweq
isweq : ∀ {ℓ} {A B : Set ℓ} (f : A → B) → Set ℓ
isweq {B = B} f = (x : B) → iscontr (f ⁻¹ x)

-- weq
weq : ∀ {ℓ} → Set ℓ → Set ℓ → Set ℓ
weq A B = Σ f ꞉ (A → B) , isweq f

-- isweqidfun
isweqid : ∀ {ℓ} {A : Set ℓ} → isweq (id {A = A})
isweqid x = (x , refl) , (λ {(x , refl) → refl})

-- idweq
idweq : ∀ {ℓ} {A : Set ℓ} → weq A A
idweq = id , isweqid

-- weqpreimage
weqpreim : ∀ {ℓ} {A B : Set ℓ} {f : A → B} (fweq : isweq f) → B → A
weqpreim fweq b with fweq b
... | (a , p) , _ = a

-- weqpreimageeq
weqpreimeq : ∀ {ℓ} {A B : Set ℓ} {f : A → B} (fweq : isweq f) (b : B)
  → f (weqpreim fweq b) ≡ b
weqpreimeq fweq b with fweq b
... | (a , p) , _ = p

-- f が weak equivalence なら、weqpreim によって f の逆関数が得られる（7.4 の内容）
