module _ {ℓ} where

  Σ' : (A : Type ℓ) (B : A → Type ℓ) → Type ℓ
  Σ' = Σ
  syntax Σ' B (λ x → E-x) = Σ x ꞉ B , E-x

  id : {A : Type ℓ} → (A → A)
  id a = a
  
  -- contractible type ((-2)-type)
  iscontr : Type ℓ → Type ℓ
  iscontr A = Σ a ꞉ A , ((x : A) → a ≡ x)

  -- proposition ((-1)-type)
  isprop : Type ℓ → Type ℓ
  isprop A = (x y : A) → x ≡ y
  -- isprop A = (x y : A) → iscontr (x ≡ y) と定めてもよい

  -- set (0-type)
  isset : Type ℓ → Type ℓ
  isset A = (x y : A) → isprop (x ≡ y)

  contr2prop : {A : Type ℓ} → iscontr A → isprop A
  contr2prop (a , f) x y = sym (f x) ∙ f y

  prop2set : {A : Type ℓ} → isprop A → isset A
  prop2set h x y p q i j =
    hcomp (λ k → λ { (i = i0) → h x (p j) k
                   ; (i = i1) → h x (q j) k
                   ; (j = i0) → h x x k
                   ; (j = i1) → h x y k})
          x
  -- HoTT でも path induction を使えば示せる

module _ {ℓ} where

  -- ⊥
  data ⊥ : Type ℓ where

  ¬ : Type ℓ → Type ℓ
  ¬ A = A → ⊥

  data Cone (A : Type ℓ) : Type ℓ where
    apex : Cone A
    base : A → Cone A
    generatrix : (x : A) → apex ≡ base x
  
  Conecontr : {A : Type ℓ} → iscontr (Cone A)
  Conecontr = apex , (λ {apex _ → apex
                       ; (base x) → generatrix x
                       ; (generatrix x i) j → generatrix x (i ∧ j)})
  
  -- Unit
  Unit = Cone ⊥
  
  Unitcontr : iscontr Unit
  Unitcontr = Conecontr

  -- Interval
  Interval = Cone Unit

  Intervalcontr : iscontr Interval
  Intervalcontr = Conecontr

  -- Bool
  data Bool : Type ℓ where
    tt ff : Bool
  
  Bool2Type : Bool → Type ℓ
  Bool2Type tt = Unit
  Bool2Type ff = ⊥

  tt≠ff : ¬ (tt ≡ ff)
  tt≠ff p = transport (λ i → Bool2Type (p i)) apex

  Boolnotprop : ¬ (isprop Bool)
  Boolnotprop Boolprop = tt≠ff (Boolprop tt ff)
  
  -- S¹ が hprop (proof irrelevant) でないことを示すために univalence を使うので、univalence を示す

  -- univalence を示す準備

  -- id は equivalence
    -- fiber が contractible であることを示せばよい

  pathspContr : {A : Type ℓ} (a : A) → isContr (Σ x ꞉ A , (x ≡ a))
  pathspContr a = (a , refl) , λ {(x , p) i → p (~ i) , λ j → p (~ i ∨ j)}
  -- id による a の fiber Σ x ꞉ A , (x ≡ a) の点として (a , refl) が取れて、
  -- fiber に属する任意の点 (x , p) は (a , refl) と等しい

  idIsEquiv : {A : Type ℓ} → isEquiv (id {A = A})
  idIsEquiv .equiv-proof a = pathspContr a

  -- library で定義されている isoToIsEquiv を使っても示せる
    -- 逆写像を作れば iso が示せて equiv も示せる
  _ : {A : Type ℓ} → isEquiv (id {A = A})
  _ = isoToIsEquiv (iso id id (λ x → refl) (λ x → refl))
  
  -- univalence の証明
  ua : {A B : Type ℓ} → A ≃ B → A ≡ B
  ua {A = A} {B} eqv i = Glue B λ { (i = i0) → A , eqv
                                  ; (i = i1) → B , id , idIsEquiv}


  not : Bool → Bool
  not tt = ff
  not ff = tt

  notinvol : (x : Bool) → not (not x) ≡ x
  notinvol tt = refl
  notinvol ff = refl

  notIsEquiv : isEquiv not
  notIsEquiv = isoToIsEquiv (iso not not notinvol notinvol)


  -- S¹
  data S¹ : Type ℓ where
    base : S¹
    loop : base ≡ base

  S¹2Type : S¹ → Type ℓ
  S¹2Type base = Bool
  S¹2Type (loop i) = ua (not , notIsEquiv) i

  -- ループの偶奇
  parity : base ≡ base → Bool
  parity p = transport (λ i → S¹2Type (p i)) tt

  _ : parity refl ≡ tt
  _ = refl

  _ : parity loop ≡ ff
  _ = refl

  _ : parity (sym loop) ≡ ff
  _ = refl

  _ : parity (loop ∙ loop) ≡ tt
  _ = refl
  -- これらが refl で示せるのは canonicity のおかげ

  refl≠loop : ¬ (refl ≡ loop)
  refl≠loop refl≡loop = tt≠ff (λ i → parity (refl≡loop i))

  S¹notset : ¬ (isset S¹)
  S¹notset S¹set = refl≠loop (S¹set base base refl loop)

  -- propositional truncation
  data ∥_∥ (A : Type ℓ) : Type ℓ where
    ∣_∣ : A → ∥ A ∥
    squash : (x y : ∥ A ∥) → x ≡ y
  
  proptruncisprop : {A : Type ℓ} → isprop ∥ A ∥
  proptruncisprop = squash

  -- （弧状）連結性 (0-connectedness)
  iscnd : Type ℓ → Type ℓ
  iscnd A = Σ a ꞉ A , ((x : A) → ∥ a ≡ x ∥)
  -- 「set truncation を取ると可縮」という定義でも良い

  S¹cnd : iscnd S¹
  S¹cnd = base , (λ {base → ∣ refl ∣
                   ; (loop i) → squash ∣ (λ j → loop (i ∧ j)) ∣ ∣ (λ j → loop (i ∨ ~ j)) ∣ i })
