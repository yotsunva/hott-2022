{-# OPTIONS --without-K --safe #-}

-- 準備

id : ∀ {ℓ} {A : Set ℓ} → (A → A)
id a = a

_∘_ : ∀ {ℓ} {A B C : Set ℓ} → (B → C) → (A → B) → (A → C)
(g ∘ f) a = g (f a)

data _≡_ {ℓ} {A : Set ℓ} (a : A) : A → Set ℓ where
  refl : a ≡ a
infix 4 _≡_

sym : ∀ {ℓ} {A : Set ℓ} {a b : A} → a ≡ b → b ≡ a
sym refl = refl

_∙_ : ∀ {ℓ} {A : Set ℓ} {a b c : A} → a ≡ b → b ≡ c → a ≡ c
refl ∙ q = q

data _×_ {ℓ} (A B : Set ℓ) : Set ℓ where
  _,_ : A → B → A × B

data Σ {ℓ} (B : Set ℓ) (E : B → Set ℓ) : Set ℓ where
  _,_ : (x : B) → E x → Σ B E
syntax Σ B (λ x → E-x) = Σ x ꞉ B , E-x

-- 5.2 functoriality (continuity)

ap : ∀ {ℓ} {A B : Set ℓ} (f : A → B) {a b : A} → a ≡ b → f a ≡ f b
ap f refl = refl

ap-id : ∀ {ℓ} {A : Set ℓ} {a b : A} {p : a ≡ b} → ap id p ≡ p
ap-id {p = refl} = refl

ap-∘ : ∀ {ℓ} {A B C : Set ℓ} {f : A → B} {g : B → C} {a b : A} {p : a ≡ b}
  → ap (g ∘ f) p ≡ ap g (ap f p)
ap-∘  {f = f} {g = g} {p = refl} = refl

ap-∙ : ∀ {ℓ} {A B : Set ℓ} {f : A → B} {a b c : A} {p : a ≡ b} {q : b ≡ c}
  → ap f (p ∙ q) ≡ (ap f p) ∙ (ap f q)
ap-∙ {p = refl} {q = refl} = refl

ap-sym : ∀ {ℓ} {A B : Set ℓ} {f : A → B} {a b c : A} {p : a ≡ b}
  → ap f (sym p) ≡ sym (ap f p)
ap-sym {p = refl} = refl

-- 6.1 homotopy

_~_ : ∀ {ℓ} {A B : Set ℓ} (f g : A → B) → Set ℓ
_~_ {A = A} f g = (x : A) → f x ≡ g x

infix 4 _~_

~refl : ∀ {ℓ} {A B : Set ℓ} {f : A → B} → f ~ f
~refl {f = f} x = refl

-- f ~ g は f, g の各点での値が等しいということ
-- 集合論においては、各点での値が等しければ写像としても等しい（写像の定義と外延性公理より）
  -- MLTT においても、外延性を公理として課せばこれは示せる（公理なしの MLTT では示せない）
  -- MLTT に公理を追加して作られる HoTT においては示せる（多分 univalence, HIT のどちらかがあれば示せる）
  -- CTT においては公理なしで示せる

isheq : ∀ {ℓ} {A B : Set ℓ} (f : A → B) → Set ℓ
isheq {A = A} {B = B} f = Σ g ꞉ (B → A) , ((f ∘ g ~ id) × (g ∘ f ~ id))

-- 実は上の定義は少しよくない（isheq は複雑な構造を持ちうる）
  -- 例えば id : S¹ → S¹ に対して、多分 isheq id id は型として Z （整数がなす型）と同値
-- 簡単な構造を持たせる（hprop にさせる）ために以下の定義を使うことが多い
  -- A ≃ B を定義して univalence を定式化するときにこの差が問題になる

isheq' : ∀ {ℓ} {A B : Set ℓ} (f : A → B) → Set ℓ
isheq' {A = A} {B = B} f = Σ g ꞉ (B → A) , Σ h ꞉ (B → A) , ((f ∘ g ~ id) × (h ∘ f ~ id))

-- transport
trf : ∀ {ℓ} {B : Set ℓ} (E : B → Set ℓ) {a b : B} → a ≡ b → (E a → E b)
trf E {a = a} refl = id

trb : ∀ {ℓ} {B : Set ℓ} (E : B → Set ℓ) {a b : B} → a ≡ b → (E b → E a)
trb E {a = a} refl = id

bf : ∀ {ℓ} {B : Set ℓ} (E : B → Set ℓ) {a b : B} → (p : a ≡ b)
  → (trb E p) ∘ (trf E p) ~ id
bf E {a = a} refl = ~refl

fb : ∀ {ℓ} {B : Set ℓ} (E : B → Set ℓ) {a b : B} → (p : a ≡ b)
  → (trf E p) ∘ (trb E p) ~ id
fb E {a = a} refl = ~refl

isheqtrf : ∀ {ℓ} {B : Set ℓ} (E : B → Set ℓ) {a b : B} → (p : a ≡ b)
  → isheq (trf E p)
isheqtrf E {a = a} refl = id , (~refl , ~refl)

isheqtrb : ∀ {ℓ} {B : Set ℓ} (E : B → Set ℓ) {a b : B} → (p : a ≡ b)
  → isheq (trb E p)
isheqtrb E {a = a} refl = id , (~refl , ~refl)
