{-# OPTIONS --without-K --safe #-}

-- id
id : ∀ {ℓ} {A : Set ℓ} → A → A
id x = x

-- composition
_∘_ : ∀ {ℓ} {A B C : Set ℓ} → (g : B → C) → (f : A → B) → (A → C)
(g ∘ f) x = g (f x)

-- identity type
data _≡_ {ℓ} {A : Set ℓ} (a : A) : A → Set ℓ where
  refl : a ≡ a

infix 4 _≡_

-- inverse
inv : ∀ {ℓ} {A : Set ℓ} {a b : A}
  → a ≡ b → b ≡ a
inv refl = refl

-- composition
_⦁_ : ∀ {ℓ} {A : Set ℓ} {a b c : A}
  → a ≡ b → b ≡ c → a ≡ c
refl ⦁ refl = refl

infix 30 _⦁_

-- unit
refl-runit : ∀ {ℓ} {A : Set ℓ} {a b : A} {p : a ≡ b}
  → (p ⦁ refl) ≡ p
refl-runit {p = refl} = refl

refl-lunit : ∀ {ℓ} {A : Set ℓ} {a b : A} {p : a ≡ b}
  → (refl ⦁ p) ≡ p
refl-lunit {p = refl} = refl

-- ap
ap : ∀ {ℓ} {A B : Set ℓ} (f : A → B) {x y}
  → x ≡ y → f x ≡ f y
ap f refl = refl

ap-fun :  ∀ {ℓ} {A B C : Set ℓ} {g : B → C} {f : A → B} {x y : A} {p : x ≡ y}
  → ap (g ∘ f) p ≡ ap g (ap f p)
ap-fun {p = refl} = refl

ap-id :  ∀ {ℓ} {A : Set ℓ} {x y : A} {p : x ≡ y}
  → ap id p ≡ p
ap-id {p = refl} = refl

ap-⦁ :  ∀ {ℓ} {A B : Set ℓ} (f : A → B) {x y z : A} {p : x ≡ y} {q : y ≡ z}
  → ap f (p ⦁ q) ≡ ap f p ⦁ ap f q
ap-⦁ f {p = refl} {q = refl} = refl

-- list
data List {ℓ} (A : Set ℓ) : Set ℓ where
  [] : List A
  _::_ : A → List A → List A

-- concatenation
_++_ : ∀ {ℓ} {A : Set ℓ} (p q : List A) → List A
[] ++ q = q
(x :: p) ++ q = x :: (p ++ q)

-- syntax sugar
_≡⟨_⟩_ : ∀ {ℓ} {A : Set ℓ} (x : A) {y z} → x ≡ y → y ≡ z → x ≡ z
x ≡⟨ p ⟩ q = p ⦁ q

_≡~⟨_⟩_ : ∀ {ℓ} {A : Set ℓ} (x : A) {y z : A} → y ≡ x → y ≡ z → x ≡ z
x ≡~⟨ p ⟩ q = (inv p) ⦁ q

_∎ : ∀ {ℓ} {A : Set ℓ} (x : A) → x ≡ x
x ∎ = refl

infixr 2 _≡⟨_⟩_ _≡~⟨_⟩_
infix  3 _∎

-- associator
assoc : ∀ {ℓ} {A : Set ℓ}
  (p : List A) (q : List A) (r : List A)
  → ((p ++ q) ++ r) ≡ (p ++ (q ++ r))
assoc [] q r = refl
assoc (x :: p) q r =
  x :: ((p ++ q) ++ r) ≡⟨ ap (x ::_) (assoc p q r) ⟩
  x :: (p ++ (q ++ r)) ∎

--pentagonator
pnt : ∀ {ℓ} {A : Set ℓ}
  (p : List A) (q : List A) (r : List A) (s : List A)
  →   (assoc (p ++ q) r s) ⦁ (assoc p q (r ++ s))
    ≡ ((ap (_++ s) (assoc p q r)) ⦁ (assoc p (q ++ r) s)) ⦁ (ap (p ++_) (assoc q r s))
pnt [] q r s =
  assoc q r s ⦁ refl ≡⟨ refl-runit ⟩
  assoc q r s ≡~⟨ ap-id ⟩
  ap id (assoc q r s) ≡~⟨ refl-lunit ⟩
  refl ⦁ ap id (assoc q r s) ∎
pnt (x :: p) q r s =
  (ap (x ::_) (assoc (p ++ q) r s) ⦁ refl)
    ⦁ (ap (x ::_) (assoc p q (r ++ s)) ⦁ refl) ≡⟨ ap (_⦁ ((ap (x ::_) (assoc p q (r ++ s)) ⦁ refl))) refl-runit ⟩
  (ap (x ::_) (assoc (p ++ q) r s))
    ⦁ (ap (x ::_) (assoc p q (r ++ s)) ⦁ refl) ≡⟨ ap ((ap (x ::_) (assoc (p ++ q) r s)) ⦁_) refl-runit ⟩
  (ap (x ::_) (assoc (p ++ q) r s))
    ⦁ (ap (x ::_) (assoc p q (r ++ s))) ≡~⟨ ap-⦁ _ ⟩
  ap (x ::_)
    ( (assoc (p ++ q) r s)
      ⦁ (assoc p q (r ++ s))) ≡⟨ ap (ap (x ::_)) (pnt p q r s) ⟩
  ap (x ::_)
    ( (ap (_++ s) (assoc p q r)
      ⦁ assoc p (q ++ r) s)
      ⦁ ap (p ++_) (assoc q r s)) ≡⟨ ap-⦁ _ ⟩
  ap (x ::_)
    ( (ap (_++ s) (assoc p q r)
      ⦁ assoc p (q ++ r) s))
    ⦁ ap (x ::_) (ap (p ++_) (assoc q r s)) ≡⟨ ap (_⦁ (ap (x ::_) (ap (p ++_) (assoc q r s)))) (ap-⦁ _) ⟩
  (ap (x ::_) (ap (_++ s) (assoc p q r))
    ⦁ ap (x ::_) (assoc p (q ++ r) s))
    ⦁ ap (x ::_) (ap (p ++_) (assoc q r s)) ≡~⟨ ap (λ z → (z ⦁ (ap (x ::_) (assoc p (q ++ r) s))) ⦁ ap (x ::_) (ap (p ++_) (assoc q r s))) ap-fun ⟩
  (ap ((x ::_) ∘ (_++ s)) (assoc p q r)
    ⦁ (ap (x ::_) (assoc p (q ++ r) s)))
    ⦁ ap (x ::_) (ap (p ++_) (assoc q r s)) ≡⟨ refl ⟩
  (ap ((_++ s) ∘ (x ::_)) (assoc p q r)
    ⦁ (ap (x ::_) (assoc p (q ++ r) s)))
    ⦁ ap (x ::_) (ap (p ++_) (assoc q r s)) ≡⟨ ap (λ z → (z ⦁ (ap (x ::_) (assoc p (q ++ r) s))) ⦁ ap (x ::_) (ap (p ++_) (assoc q r s))) ap-fun ⟩
  (ap (_++ s) (ap (x ::_) (assoc p q r))
    ⦁ (ap (x ::_) (assoc p (q ++ r) s)))
    ⦁ ap (x ::_) (ap (p ++_) (assoc q r s)) ≡~⟨ ap ((ap (_++ s) (ap (x ::_) (assoc p q r)) ⦁ (ap (x ::_) (assoc p (q ++ r) s))) ⦁_) ap-fun ⟩
  (ap (_++ s) (ap (x ::_) (assoc p q r))
    ⦁ (ap (x ::_) (assoc p (q ++ r) s)))
    ⦁ ap (λ t → x :: (p ++ t)) (assoc q r s) ≡~⟨ ap (λ z → (ap (_++ s) (ap (x ::_) (assoc p q r)) ⦁ z) ⦁ ap (λ t → x :: (p ++ t)) (assoc q r s)) refl-runit ⟩
  (ap (_++ s) (ap (x ::_) (assoc p q r))
    ⦁ (ap (x ::_) (assoc p (q ++ r) s) ⦁ refl))
    ⦁ ap (λ t → x :: (p ++ t)) (assoc q r s) ≡~⟨ ap (λ z → ((ap (_++ s) z) ⦁ (ap (x ::_) (assoc p (q ++ r) s) ⦁ refl)) ⦁ ap (λ t → x :: (p ++ t)) (assoc q r s)) refl-runit ⟩
  (ap (_++ s) (ap (x ::_) (assoc p q r) ⦁ refl)
    ⦁ (ap (x ::_) (assoc p (q ++ r) s) ⦁ refl))
    ⦁ ap (λ t → x :: (p ++ t)) (assoc q r s) ∎
