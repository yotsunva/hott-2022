{-# OPTIONS --without-K --safe #-}

data _≡_ {ℓ} {A : Set ℓ} (a : A): A → Set ℓ where
  refl : a ≡ a

-- 合成（順番が標準的なものと異なるかも）
_⦁_ : ∀ {ℓ} {A : Set ℓ} {a b c : A}
  → a ≡ b → b ≡ c → a ≡ c
refl ⦁ refl = refl

infix 30 _⦁_

-- 水平合成（順番が標準的なものと異なるかも）
_∘_ : ∀ {ℓ} {A : Set ℓ} {a b c : A} {p q : a ≡ b} {r s : b ≡ c}
  → p ≡ q → r ≡ s → (p ⦁ r ≡ q ⦁ s)
refl ∘ refl = refl

-- associativity up to homotopy
assoc : ∀ {ℓ} {A : Set ℓ} {a b c d : A}
  (p : a ≡ b) (q : b ≡ c) (r : c ≡ d)
  → ((p ⦁ q) ⦁ r) ≡ (p ⦁ (q ⦁ r))
assoc refl refl refl = refl

-- associativity up to coherent homotopy は無数のホモトピーからなるデータ
-- assoc や pnt はそのうちの一つ
pnt : ∀ {ℓ} {A : Set ℓ} {a b c d e : A}
  (p : a ≡ b) (q : b ≡ c) (r : c ≡ d) (s : d ≡ e)
  → ( (assoc (p ⦁ q) r s) ⦁ (assoc p q (r ⦁ s))
      ≡ ((assoc p q r ∘ refl) ⦁ (assoc p (q ⦁ r) s)) ⦁ (refl ∘ assoc q r s))
pnt refl refl refl refl = refl

pnt' : ∀ {ℓ} {A : Set ℓ} {a b c d e : A}
  (p : a ≡ b) (q : b ≡ c) (r : c ≡ d) (s : d ≡ e)
  → ( (assoc (p ⦁ q) r s) ⦁ (assoc p q (r ⦁ s))
      ≡ (assoc p q r ∘ refl) ⦁ ((assoc p (q ⦁ r) s) ⦁ (refl ∘ assoc q r s)))
pnt' refl refl refl refl = refl
