{-# OPTIONS --without-K --safe #-}

-- 準備

id : ∀ {ℓ} {A : Set ℓ} → (A → A)
id a = a

_∘_ : ∀ {ℓ} {A B C : Set ℓ} → (B → C) → (A → B) → (A → C)
(g ∘ f) a = g (f a)

data _≡_ {ℓ} {A : Set ℓ} (a : A) : A → Set ℓ where
  refl : a ≡ a
infix 4 _≡_

data ⊥ : Set where

¬ : ∀ {ℓ} → Set ℓ → Set ℓ
¬ A = A → ⊥

data Unit : Set where
  star : Unit

data Bool : Set where
  tt ff : Bool

data Σ {ℓ} (B : Set ℓ) (E : B → Set ℓ) : Set ℓ where
  _,_ : (x : B) → E x → Σ B E
syntax Σ B (λ x → E-x) = Σ x ꞉ B , E-x

pr1 : ∀ {ℓ} {B : Set ℓ} {E : B → Set ℓ} → (Σ x ꞉ B , E x) → B
pr1 (x , y) = x

pr2 : ∀ {ℓ} {B : Set ℓ} {E : B → Set ℓ} → (p : Σ x ꞉ B , E x) → E (pr1 p)
pr2 (x , y) = y

_~_ : ∀ {ℓ} {A B : Set ℓ} (f g : A → B) → Set ℓ
_~_ {A = A} f g = (x : A) → f x ≡ g x
infix 4 _~_

trf : ∀ {ℓ} {B : Set ℓ} (E : B → Set ℓ) {a b : B} → a ≡ b → (E a → E b)
trf E {a = a} refl x = x


-- 6.3  Paths in the total space

ptot : ∀ {ℓ} {B : Set ℓ} {E : B → Set ℓ} {x y : Σ b ꞉ B , E b}
  → (p : pr1 x ≡ pr1 y) → (trf E p (pr2 x) ≡ pr2 y) → x ≡ y
ptot {x = a , x} {y = .a , .x} refl refl = refl

ptotpr1 : ∀ {ℓ} {B : Set ℓ} {E : B → Set ℓ} {x y : Σ b ꞉ B , E b}
  → x ≡ y →  pr1 x ≡ pr1 y
ptotpr1 refl = refl

ptotpr2 : ∀ {ℓ} {B : Set ℓ} {E : B → Set ℓ} {x y : Σ b ꞉ B , E b}
  → (p : x ≡ y) → trf E (ptotpr1 p) (pr2 x) ≡ pr2 y
ptotpr2 refl = refl

ptotchar : ∀ {ℓ} {B : Set ℓ} {E : B → Set ℓ} {x y : Σ b ꞉ B , E b}
  → (p : x ≡ y) → p ≡ ptot (ptotpr1 p) (ptotpr2 p)
ptotchar {x = a , x} refl = refl


-- 7.1 Contractibility

iscontr : ∀ {ℓ} → Set ℓ → Set ℓ
iscontr A = Σ a ꞉ A , ((x : A) → a ≡ x)

-- propositional proof irrelevance（これを満たす型を h-proposition と言ったりする）
  -- HoTT の文脈では h-proposition のことを単に proposition と言うこともある
isprop : ∀ {ℓ} → Set ℓ → Set ℓ
isprop A = (x y : A) → x ≡ y

contr2prop : ∀ {ℓ} {A : Set ℓ} → iscontr A → isprop A
contr2prop (a , f) x y with f x | f y
...  | refl | refl = refl

Unitcontr : iscontr Unit
Unitcontr = star , λ {star → refl}

⊥notcontr : ¬ (iscontr ⊥)
⊥notcontr (a , f) = a

⊥prop : isprop ⊥
⊥prop () ()

Boolnotprop : ¬ (isprop Bool)
Boolnotprop Boolprop with Boolprop tt ff
...  | ()

Boolnotcontr : ¬ (iscontr Bool)
Boolnotcontr x = Boolnotprop (contr2prop x)

iscontrret : ∀ {ℓ} {A B : Set ℓ} (r : A → B) (s : B → A)
  → r ∘ s ~ id → iscontr A → iscontr B
iscontrret {B = B} r s H (a , f) = r a , lemma
  where
  lemma : (x : B) → r a ≡ x
  lemma x with f (s x)
  ...  | refl = H x
