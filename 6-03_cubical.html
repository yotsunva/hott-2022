data _×_ {ℓ} (A B : Set ℓ) : Set ℓ where
  _,_ : A → B → A × B

Σ' : ∀ {ℓ} (A : Set ℓ) (B : A → Set ℓ) → Set ℓ
Σ' = Σ
syntax Σ' B (λ x → E-x) = Σ x ꞉ B , E-x


module _ {ℓ} where

  id : {A : Set ℓ} → (A → A)
  id a = a

  _∘_ : {A B C : Set ℓ} → (B → C) → (A → B) → (A → C)
  (g ∘ f) a = g (f a)

  -- 5.2 functoriality (continuity)
  ap : {A B : Set ℓ} (f : A → B) {a b : A} → a ≡ b → f a ≡ f b
  ap f p = λ i → f (p i) 
  
  -- 6.1 homotopy
  module _ {A B : Set ℓ} where

    _~_ : (f g : A → B) → Set ℓ
    _~_ f g = (x : A) → f x ≡ g x
  
    infix 4 _~_

    -- CTT, HoTT においては、f ~ g から f ≡ g が従う
      -- MLTT では各点における値が等しくても関数として等しいことは一般には示せない
    funext : {f g : A → B} → (f ~ g) → (f ≡ g)
    funext = λ H i x → H x i
  
  module _ {A B : Set ℓ} where

    -- isheq' : (f : A → B) → Set ℓ
    -- isheq' f = Σ g ꞉ (B → A) , ((f ∘ g ~ id) × (g ∘ f ~ id))

    -- 以下のように定義した方が良くふるまう（構造が単純になる）
    isheq : (f : A → B) → Set ℓ
    isheq f = Σ g ꞉ (B → A) , Σ h ꞉ (B → A) , ((f ∘ g ~ id) × (h ∘ f ~ id))

  -- 型の同値性
  _≃_ : (A B : Set ℓ) → Set ℓ
  A ≃ B = Σ f ꞉ (A → B) , isheq f
  -- 可縮性を使う定義もある



  -- HIT の話
  -- 主に以下の話題を扱う
    -- Unit ≃ Interval
    -- Unit ≃ Cone A
      -- これは Unit ≃ Interval の一般化
      -- Unit ≃ D² も従う
    -- T² ≃ (S¹ × S¹)
    -- Klein bottle, real projective plane の定義
    -- {(x , y) ∈ R² : x² + y² ≦ 1} と D² の関係


  -- unit type
  data Unit : Set ℓ where
    star : Unit

  -- interval
  data Interval : Set ℓ where
    I0 I1 : Interval
    seg : I0 ≡ I1
  
  
  
  constI0 : Interval → Interval
  constI0 x = I0

  constI0~id : constI0 ~ id
  constI0~id I0 = refl
  constI0~id I1 = seg
  constI0~id (seg i) = λ j → seg (i ∧ j)
  -- 端点だけでなく、seg 上の点に対しても path を与える必要がある
  
  
  
  -- Unit, Interval は型として同値
  _ : Unit ≃ Interval
  _ = (λ x → I0) , (λ i → star) , (λ i → star) , constI0~id , λ {star → refl}
  
  data Cone (A : Set ℓ) : Set ℓ where
    apex : Cone A
    base : A → Cone A
    generatrix : (x : A) → apex ≡ base x
  
  
  
  -- Interval を Cone Unit として定めることもある
  -- Cone A は id : A → A, const : A → Unit の pushout として定めることもある

  module _ {A : Set ℓ} where

    constapex : Cone A → Cone A
    constapex x = apex

    constapex~id : constapex ~ id
    constapex~id apex = refl
    constapex~id (base x) = generatrix x
    constapex~id (generatrix x i) = λ j → generatrix x (i ∧ j)

    -- Unit, Cone A は型として同値
    Unit≃Cone : Unit ≃ Cone A
    Unit≃Cone = (λ x → apex) , (λ c → star) , (λ c → star) , constapex~id , λ {star → refl}

  -- circle
  data S¹ : Set ℓ where
    base : S¹
    loop : base ≡ base
  
  -- disc
  D² = Cone S¹

  

  _ : Unit ≃ D²
  _ = Unit≃Cone

  data T² : Set ℓ where
    pt : T²
    l1 l2 : pt ≡ pt
    sq : PathP (λ i → l1 i ≡ l1 i) l2 l2

  t2c : T² → S¹ × S¹
  t2c pt = base , base
  t2c (l1 i) = loop i , base
  t2c (l2 i) = base , loop i
  t2c (sq i j) = loop i , loop j

  

  c2t : S¹ × S¹ → T²
  c2t (base , base) = pt
  c2t (loop i , base) = l1 i
  c2t (base , loop i) = l2 i
  c2t (loop i , loop j) = sq i j
  
  c2t-t2c : c2t ∘ t2c ~ id
  c2t-t2c pt = refl
  c2t-t2c (l1 i) = refl
  c2t-t2c (l2 i) = refl
  c2t-t2c (sq i j) = refl

  t2c-c2t : t2c ∘ c2t ~ id
  t2c-c2t (base , base) = refl
  t2c-c2t (base , loop i) = refl
  t2c-c2t (loop i , base) = refl
  t2c-c2t (loop i , loop j) = refl

  T²≃S¹×S¹ : T² ≃ (S¹ × S¹)
  T²≃S¹×S¹ = t2c , c2t , c2t , t2c-c2t , c2t-t2c
  -- これの証明は HoTT だともう少し面倒なはず
    -- CTT だとたくさん簡約できるので、c2t-t2c, t2c-c2t の証明が楽

  -- torus と同様に定義できる型

  -- Klein bottle
  data K² : Set ℓ where
    pt : K²
    l1 l2 : pt ≡ pt
    sq : PathP (λ i → l1 (~ i) ≡ l1 i) l2 l2

  

  -- real projective plane
  data RP² : Set ℓ where
    pt : RP²
    l1 l2 : pt ≡ pt
    sq : PathP (λ i → l1 (~ i) ≡ l1 i) l2 (sym l2)
