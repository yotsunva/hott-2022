{-# OPTIONS --without-K --safe #-}

data ℕ : Set where
  zero : ℕ
  suc  : ℕ → ℕ

{-# BUILTIN NATURAL ℕ #-}

data _≡_ {ℓ} {A : Set ℓ} (a : A) : A → Set ℓ where
  refl : a ≡ a
infix 4 _≡_

_⁻¹ : ∀ {ℓ} {A : Set ℓ} {a b : A} → a ≡ b → b ≡ a
refl ⁻¹ = refl

_∙_ : ∀ {ℓ} {A : Set ℓ} {a b c : A} → a ≡ b → b ≡ c → a ≡ c
refl ∙ p = p
infix 10 _∙_

linv : ∀ {ℓ} {A : Set ℓ} {a b : A} (p : a ≡ b)
  → p ⁻¹ ∙ p ≡ refl
linv refl = refl

record Σ {ℓ} (B : Set ℓ) (E : B → Set ℓ) : Set ℓ where
  constructor _,_
  field
    fst : B
    snd : E fst
open Σ public
infix 4 _,_
syntax Σ B (λ x → E-x) = Σ x ꞉ B , E-x

iscontr : ∀ {ℓ} → Set ℓ → Set ℓ
iscontr A = Σ a ꞉ A , ((x : A) → a ≡ x)

isprop : ∀ {ℓ} → Set ℓ → Set ℓ
isprop A = (x y : A) → x ≡ y

isset : ∀ {ℓ} → Set ℓ → Set ℓ
isset A = (x y : A) → isprop (x ≡ y)

isOfHLevel : ∀ {ℓ} → ℕ → Set ℓ → Set ℓ
isOfHLevel zero    A = iscontr A
isOfHLevel (suc n) A = (x y : A) → isOfHLevel n (x ≡ y)

isOfHLevelsuc : ∀ {ℓ} (n : ℕ) (A : Set ℓ)
  → isOfHLevel n A → isOfHLevel (suc n) A
isOfHLevelsuc zero A (a , f) x  y .fst      = ((f x)⁻¹ ∙ f y)
isOfHLevelsuc zero A (a , f) x .x .snd refl = linv (f x)
isOfHLevelsuc (suc n) A f x y = isOfHLevelsuc n (x ≡ y) (f x y)

funext : ∀ {ℓ} → Set (Agda.Primitive.lsuc ℓ)
funext {ℓ} = {A : Set ℓ} {B : A → Set ℓ} → {f g : (x : A) → B x} → ((x : A) → f x ≡ g x) → f ≡ g
