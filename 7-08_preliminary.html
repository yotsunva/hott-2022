{-# OPTIONS --without-K --safe #-}

data _≡_ {ℓ} {A : Set ℓ} (a : A) : A → Set ℓ where
  refl : a ≡ a
infix 4 _≡_

_⁻¹ : ∀ {ℓ} {A : Set ℓ} {a b : A} → a ≡ b → b ≡ a
refl ⁻¹ = refl

_∙_ : ∀ {ℓ} {A : Set ℓ} {a b c : A} → a ≡ b → b ≡ c → a ≡ c
refl ∙ p = p
infix 10 _∙_

record Σ {ℓ} (B : Set ℓ) (E : B → Set ℓ) : Set ℓ where
  constructor _,_
  field
    fst : B
    snd : E fst
open Σ public
infix 4 _,_
syntax Σ B (λ x → E-x) = Σ x ꞉ B , E-x

iscontr : ∀ {ℓ} → Set ℓ → Set ℓ
iscontr A = Σ a ꞉ A , ((x : A) → a ≡ x)

isprop : ∀ {ℓ} → Set ℓ → Set ℓ
isprop A = (x y : A) → x ≡ y

isset : ∀ {ℓ} → Set ℓ → Set ℓ
isset A = (x y : A) → isprop (x ≡ y)

iscontr→isprop : ∀ {ℓ} {A : Set ℓ} → iscontr A → isprop A
iscontr→isprop {A = A} (a , f) x y = (f x)⁻¹ ∙ f y

isprop→isset : ∀ {ℓ} {A : Set ℓ} → isprop A → isset A
isprop→isset {A = A} f x y p q = lemma₁ p ∙ (lemma₁ q)⁻¹
  where
  lemma₁ : (r : x ≡ y) → r ≡ f x y
  lemma₁ r = lemma₂ x r (f x x) refl
  -- r と f x x が refl である場合に帰着させたい
  -- path induction を使える形になるように命題を一般化する
  -- 一般化された命題 lemma₂ を示した後、lemma₂ に r と f x x を代入する 
    where
    lemma₂ : (z : A) → (s : x ≡ y) → (t : x ≡ z) → t ≡ f x z → s ≡ f x y
    lemma₂ _ refl refl α = α

funext : ∀ {ℓ} → Set (Agda.Primitive.lsuc ℓ)
funext {ℓ} = {A : Set ℓ} {B : A → Set ℓ} → {f g : (x : A) → B x} → ((x : A) → f x ≡ g x) → f ≡ g
