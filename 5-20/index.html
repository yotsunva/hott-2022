-- 5.1. paths

-- (based) identity type（こちらを使うことにする）
data Id {ℓ} {A : Set ℓ} (a : A) : A → Set ℓ where
  refl : Id a a

-- unbased identity type（今回は使わない。based なものと型として同値だが、除去則が少し異なる（HoTT book 1.12.1 を参照））
data ubId {ℓ} {A : Set ℓ} : A → A → Set ℓ where
  refl : {a : A} → ubId a a

-- inverse
inv : ∀ {ℓ} {A : Set ℓ} {a b : A} → Id a b → Id b a
inv refl = refl

-- 余談
-- MLTT において、Id a _ の canonical term は refl : Id a a のみである
-- Id a _ からの関数を作りたければ、canonical term の行き先を定めればよい（これが除去則の内容）
-- 「閉項を canonical term に簡約できる」という canonicity は公理を付けると崩れる
  -- HoTT における higher inductive type S¹ について、base : S¹, loop : Id base base という閉項が作れるが、loop は refl に簡約できない
  -- HoTT においては、このような inv の定義だと refl の行き先しか定めていないので、inv loop や inv (inv loop) などはこれ以上簡約できない
  -- canonical term に簡約できない項がたくさん作れてしまう
    -- 例えば、ある条件を満たす nat の閉項が作れたときに、それを suc (suc ... (0) ... ) という形に簡約できないかもしれない
    -- つまり、ある意味で議論の構成性が崩れてしまう
-- MLTT の規則を少し改変した cubical type theory (CTT) において、この問題は一応解決される
  -- Id のような inductive family の canonical term として、refl 以外のものが追加される（cubical agda の表記を使うと、transpX ... というもの）
  -- Id _ _ の閉項は refl か transpX ... のどちらかの形になるよう簡約できる (canonicity)
    -- CTT において、S¹ の loop は Path base base という型を持っていて、この loop から transpX を使って Id base base の項が作れる
  -- CTT においては、inv を定めるときに transpX ... の行き先も定めることになる
    -- そのおかげで inv (transpX ...) の形になっても簡約を続けられる

-- composition
comp : ∀ {ℓ} {A : Set ℓ} {a b c : A} → Id a b → Id b c → Id a c
comp refl q = q

-- 他の定義も考えられる
comp₁ : ∀ {ℓ} {A : Set ℓ} {a b c : A} → Id a b → Id b c → Id a c
comp₁ p refl = p

comp₂ : ∀ {ℓ} {A : Set ℓ} {a b c : A} → Id a b → Id b c → Id a c
comp₂ refl refl = refl

-- どれもある意味では等しい（definitionally equal (judgmentally equal) ではないが、propositionally equal）
comp-comp₁ : ∀ {ℓ} {A : Set ℓ} {a b c : A} (p : Id a b) (q : Id b c) → Id (comp p q) (comp₁ p q)
comp-comp₁ refl refl = refl

comp-comp₂ : ∀ {ℓ} {A : Set ℓ} {a b c : A} (p : Id a b) (q : Id b c) → Id (comp p q) (comp₂ p q)
comp-comp₂ refl refl = refl

-- だが、簡約できるかどうかが変わってくる
-- q が変数であるとき、comp refl q は q に簡約できるが、comp₁ refl q はこれ以上簡約できない
-- 例えば、Id (comp refl q) q の証明は refl で済むが、Id (comp₁ refl q) q の証明では q に関するパターンマッチを行う必要がある

syntax comp p q = p ⋅ q

refl-lunit : ∀ {ℓ} {A : Set ℓ} {a b : A} (p : Id a b) → Id (refl ⋅ p) p
refl-lunit p = refl
-- comp の定義より、refl ⋅ p は p に簡約される
-- Id p p は refl で示せる

refl-runit : ∀ {ℓ} {A : Set ℓ} {a b : A} (p : Id a b) → Id (p ⋅ refl) p
refl-runit refl = refl
-- p ⋅ refl はこれ以上簡約できないので、p に関するパターンマッチ（Id の除去則と同等のもの）を使う必要がある
-- lunit の方もパターンマッチを行っても良い（comp₂ を採用するとどちらの場合もパターンマッチが必要）


-- いくつか基本的な命題を挙げる（どれも可能な限りパターンマッチを行えば、後は refl で示せる）
inv-inv : ∀ {ℓ} {A : Set ℓ} {a b : A} (p : Id a b) → Id (inv (inv p)) p
inv-inv refl = refl

comp-assoc : ∀ {ℓ} {A : Set ℓ} {a b c d : A} (p : Id a b) (q : Id b c) (r : Id c d) →
  Id ((p ⋅ q) ⋅ r) (p ⋅ (q ⋅ r))
comp-assoc refl refl refl = refl

inv-linv : ∀ {ℓ} {A : Set ℓ} {a b c : A} (p : Id a b) (q : Id b c) → Id ((inv p) ⋅ (p ⋅ q)) q
inv-linv refl refl = refl

inv-rinv : ∀ {ℓ} {A : Set ℓ} {a b c : A} (p : Id a b) (q : Id b c) → Id ((p ⋅ q) ⋅ (inv q)) p
inv-rinv refl refl = refl


-- MLTT のホモトピー論的なモデルについて考えると、refl 以外の path も色々ある
-- refl についてしか示していないので不安になるが、上の証明で実際にモデル側でも対応する命題が成り立つことが保証される
  -- 不安になるのは、
    -- ホモトピー的なモデルがちゃんとモデルになっている
    -- （特に Id の除去則に対応する命題がホモトピー論的なモデルにおいて成り立っている）
  -- という事実が非自明だからかもしれない（Theorem 2.1 辺りでやった内容）
-- 構文について考えると、（HoTT において）これらの証明を loop : Id base base のような閉項に適用すると簡約が進まないという不安がある
  -- 上で述べたように、refl 以外の閉項の存在を認めつつこれを解決するには規則を変える必要がある
